linkedlist* ll_construct()
	Constructs a linked list.
	Returns a pointer to a linkedlist struct.

void ll_destruct(linkedlist* list)
	Deconstructs a linkedlist.

void ll_add(linkedlist* list, void* data)
	Adds a void pointer to the end of the linkedlist.

int ll_size(linkedlist* list)
	Returns the size of the linkedlist.

void ll_clear(linkedlist* list)
	Releases all memory required for the nodes in the linkedlist.
